Переменные находятся в блоке "if" за его пределами видно только переменную объявленную через "var"

Чтобы не было ошибок нужно вывод в консоль для переменных объявленных через let и const поместить в блок "if" или объявить их через "var"

function scopeTest() {
   if (true) {
      var testVar = 'var variable';
      let blockVar = 'let variable';
      const blockConst = 'const variable';
      console.log(blockVar);
      console.log(blockConst);
   }
   console.log(testVar);
}
scopeTest();
